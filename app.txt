import streamlit as st
from ultralytics import YOLO
import tempfile
import os

st.set_page_config(page_title="Detector de Eucaliptos", page_icon="🌳")
st.title("🌳 Detector de Eucaliptos con IA")
st.write("Subí una foto y la IA contará los objetos automáticamente.")

# Cargar el modelo
@st.cache_resource
def cargar_modelo():
    return YOLO("yolov8n.pt")

modelo = cargar_modelo()

# Subir imagen
archivo = st.file_uploader("📸 Subí una foto", type=["jpg", "jpeg", "png"])

if archivo is not None:
    # Guardar temporal
    with tempfile.NamedTemporaryFile(delete=False, suffix=".jpg") as tmp:
        tmp.write(archivo.getbuffer())
        ruta_tmp = tmp.name
    
    # Mostrar original
    st.image(archivo, caption="Imagen original", use_container_width=True)
    
    # Procesar
    with st.spinner("🔍 Analizando..."):
        resultados = modelo(ruta_tmp)
    
    # Mostrar resultado
    imagen_resultado = resultados[0].plot()
    st.image(imagen_resultado, caption="Detecciones", use_container_width=True)
    
    # Contar
    cantidad = len(resultados[0].boxes)
    st.success(f"✅ Se detectaron {cantidad} objetos")
    
    # Limpiar
    os.unlink(ruta_tmp)
